function calculate() { const volume = getCenterValue(volumeList); const sh = getCenterValue(shList); const surfactant = getCenterValue(surfactantList); const scent = getCenterValue(scentList); if ([volume, sh, surfactant, scent].includes(null)) return; const { stockSH, shCostPerL, surfCostPerL, scentCostPerL } = loadSettings(); const shVol = (sh / stockSH) * volume; const surfVol = surfactant * volume; const scentVol = scent * volume; const water = volume - shVol - surfVol - scentVol; const shCost = shVol * shCostPerL; const surfCost = surfVol * surfCostPerL; const scentCost = scentVol * scentCostPerL; // Update only the necessary fields document.getElementById("waterVol").textContent = `${water.toFixed(2)}L`; document.getElementById("shVol").textContent = `${shVol.toFixed(2)}L`; document.getElementById("surfVol").textContent = `${surfVol.toFixed(2)}L`; document.getElementById("scentVol").textContent = `${scentVol.toFixed(2)}L`; document.getElementById("shCost").textContent = `$${shCost.toFixed(2)}`; document.getElementById("surfCost").textContent = `$${surfCost.toFixed(2)}`; document.getElementById("scentCost").textContent = `$${scentCost.toFixed(2)}`; }
